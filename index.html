Cloud Storage Configuration

To enable large file uploads (up to 200MB), you need to configure your server to handle large files. Here's the configuration you should implement:

        
// server.js
const express = require('express');
const multer = require('multer');
const path = require('path');
const app = express();
// Configure storage for large files
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/');
  },
  filename: (req, file, cb) => {
    cb(null, Date.now() + path.extname(file.originalname));
  }
});
const upload = multer({
  storage: storage,
  limits: { fileSize: 200 * 1024 * 1024 } // 200MB limit
});
// Upload endpoint
app.post('/upload', upload.single('image'), (req, res) => {
  if (!req.file) {
    return res.status(400).json({ error: 'No file uploaded' });
  }
  // Return cloud URL
  res.json({
    url: `/uploads/${req.file.filename}`,
    size: req.file.size
  });
});
// In your admin dashboard
async function uploadLargeImage(file) {
  const formData = new FormData();
  formData.append('image', file);
  const response = await fetch('/upload', {
    method: 'POST',
    body: formData
  });
  const result = await response.json();
  return result.url;
}

This configuration uses multer to handle file uploads and sets a 200MB file size limit. The uploaded files will be stored in the 'uploads/' directory.
